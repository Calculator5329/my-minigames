### `lib/gates.js`

```js
// Gate catalog loader and evaluator. Primitive combinational gates
// expose an eval(a, b?) -> y function. IO / const / custom gates are
// identified by `kind` and handled specially by the sim.
(function (root, factory) {
  const mod = factory();
  if (typeof module !== 'undefined' && module.exports) module.exports = mod;
  if (typeof window !== 'undefined') {
    window.NDP = window.NDP || {};
    window.NDP.Sand = window.NDP.Sand || {};
    window.NDP.Sand.Gates = mod.Gates;
  }
})(typeof self !== 'undefined' ? self : this, function () {
  const EVAL = {
    NOT:  (a)    => a ? 0 : 1,
    AND:  (a, b) => (a && b) ? 1 : 0,
    OR:   (a, b) => (a || b) ? 1 : 0,
    NAND: (a, b) => (a && b) ? 0 : 1,
    NOR:  (a, b) => (a || b) ? 0 : 1,
    XOR:  (a, b) => (a ^ b) & 1,
    XNOR: (a, b) => (a ^ b) ? 0 : 1
  };
  const Gates = {
    loadPrimitives(json) { Gates.primitives = json.primitives; return Gates.primitives; },
    get(type) { return Gates.primitives && Gates.primitives[type]; },
    evalCombo(type, a, b) {
      const fn = EVAL[type];
      if (!fn) throw new Error('unknown combo gate: ' + type);
      return fn(a | 0, b | 0);
    },
    isCombo(type) { return !!EVAL[type]; }
  };
  return { Gates };
});
```

### `lib/sim.js`

```js
/* Gate-level simulator.

   Circuit shape:
     { nodes: [ { id, type, props? } ], wires: [ { from: {node,pin}, to: {node,pin} } ] }

   Sequential types that carry state across ticks:
     DLATCH, DFF, SRLATCH, CUSTOM (if its impl contains sequential)
   For now only primitive combinational types + IO + CONST are first-class;
   sequential are implemented as custom components in the level data.

   build(circuit) -> compiled graph with { nodes, wires, order, seqIds }
   tick(graph, inputsMap) -> outputsMap
   reset(graph) -- clears cached values / seq state
*/
(function (root, factory) {
  const mod = factory();
  if (typeof module !== 'undefined' && module.exports) module.exports = mod;
  if (typeof window !== 'undefined') {
    window.NDP = window.NDP || {};
    window.NDP.Sand = window.NDP.Sand || {};
    window.NDP.Sand.Sim = mod.Sim;
  }
})(typeof self !== 'undefined' ? self : this, function () {
  const SEQ_TYPES = new Set(['DLATCH', 'DFF', 'SRLATCH']);
  const COMBO_FN = {
    NOT:  (a)    => a ? 0 : 1,
    AND:  (a, b) => (a && b) ? 1 : 0,
    OR:   (a, b) => (a || b) ? 1 : 0,
    NAND: (a, b) => (a && b) ? 0 : 1,
    NOR:  (a, b) => (a || b) ? 0 : 1,
    XOR:  (a, b) => (a ^ b) & 1,
    XNOR: (a, b) => (a ^ b) ? 0 : 1
  };

  function build(circuit) {
    const nodes = new Map();
    for (const n of circuit.nodes) nodes.set(n.id, Object.assign({}, n, { in: {}, out: {}, state: {} }));
    const wiresByDst = new Map();   // "nodeId.pin" -> [ { fromNode, fromPin } ]
    for (const w of circuit.wires) {
      const key = w.to.node + '.' + w.to.pin;
      const list = wiresByDst.get(key) || [];
      list.push({ fromNode: w.from.node, fromPin: w.from.pin });
      wiresByDst.set(key, list);
    }
    // Topo sort ignoring edges INTO seq nodes (those are the state cut).
    const adj = new Map();  // nodeId -> Set of nodeIds this depends on
    for (const n of circuit.nodes) adj.set(n.id, new Set());
    for (const w of circuit.wires) {
      if (SEQ_TYPES.has(nodes.get(w.to.node).type)) continue;
      adj.get(w.to.node).add(w.from.node);
    }
    const order = [];
    const temp = new Set();
    const perm = new Set();
    function visit(id) {
      if (perm.has(id)) return;
      if (temp.has(id)) throw new Error('combinational cycle at ' + id);
      temp.add(id);
      for (const dep of adj.get(id)) visit(dep);
      temp.delete(id); perm.add(id); order.push(id);
    }
    for (const id of adj.keys()) visit(id);
    const seqIds = [];
    for (const [id, n] of nodes) if (SEQ_TYPES.has(n.type)) seqIds.push(id);
    return { nodes, wiresByDst, order, seqIds };
  }

  function reset(g) {
    for (const n of g.nodes.values()) { n.out = {}; n.in = {}; n.state = {}; }
  }

  function readPin(g, nodeId, pin, prevStateOnly) {
    const wires = g.wiresByDst.get(nodeId + '.' + pin);
    if (!wires || !wires.length) return 0;
    const w = wires[0];   // single-driver assumption
    const src = g.nodes.get(w.fromNode);
    if (prevStateOnly && SEQ_TYPES.has(src.type)) return (src.state.q | 0);
    return (src.out[w.fromPin] | 0);
  }

  function tick(g, inputsMap) {
    // 1. Prime INPUT / CONST / CLOCK output values.
    for (const n of g.nodes.values()) {
      if (n.type === 'INPUT')  n.out.y = (inputsMap[n.props && n.props.label] || 0) & 1;
      if (n.type === 'CONST0') n.out.y = 0;
      if (n.type === 'CONST1') n.out.y = 1;
      if (n.type === 'CLOCK')  n.out.y = (inputsMap.__clk || 0) & 1;
    }
    // 2. Combinational evaluate in topo order.
    for (const id of g.order) {
      const n = g.nodes.get(id);
      if (n.type === 'INPUT' || n.type === 'CONST0' || n.type === 'CONST1' || n.type === 'CLOCK') continue;
      if (COMBO_FN[n.type]) {
        const a = readPin(g, id, 'a', true);
        const b = readPin(g, id, 'b', true);
        n.out.y = n.type === 'NOT'
          ? COMBO_FN.NOT(a)
          : COMBO_FN[n.type](a, b);
      } else if (n.type === 'OUTPUT') {
        n.out.value = readPin(g, id, 'a', true);
      }
      // Sequential handled in step 3.
    }
    // 3. Sequential capture/commit.
    for (const id of g.seqIds) {
      const n = g.nodes.get(id);
      const prevQ = n.state.q | 0;
      let nextQ = prevQ;
      if (n.type === 'DFF') {
        const clk = readPin(g, id, 'clk', false) | 0;
        const d   = readPin(g, id, 'd',   false) | 0;
        if (clk && !n.state.lastClk) nextQ = d;   // rising-edge capture
        n.state.lastClk = clk;
      } else if (n.type === 'DLATCH') {
        const en = readPin(g, id, 'en', false) | 0;
        const d  = readPin(g, id, 'd',  false) | 0;
        if (en) nextQ = d;
      } else if (n.type === 'SRLATCH') {
        const s = readPin(g, id, 's', false) | 0;
        const r = readPin(g, id, 'r', false) | 0;
        if (s && !r) nextQ = 1;
        else if (r && !s) nextQ = 0;
      }
      n.state.q = nextQ;
      n.out.q = nextQ;
      n.out.qn = nextQ ? 0 : 1;
    }
    // 4. Collect OUTPUT labels.
    const out = {};
    for (const n of g.nodes.values()) {
      if (n.type === 'OUTPUT') out[n.props && n.props.label] = (n.out.value | 0);
    }
    return out;
  }

  return { Sim: { build, reset, tick } };
});
```

## Task 15 — Encoding + Comparison + Sequential + ALU levels (15 files)

**Files:**
- Create: 3 encoding, 3 comparison, 5 sequential, 4 ALU level files (per design).

Sequential level files note a `"sequential": true` flag so the level runner toggles the clock around each truth-table row.

Sequential truth-table encoding:

```json
"truthTable": [
  { "in": { "D": 0, "__clk": "rise" }, "out": { "Q": 0 } },
  { "in": { "D": 1, "__clk": "rise" }, "out": { "Q": 1 } },
  { "in": { "D": 0, "__clk": "stay" }, "out": { "Q": 1 } },
  { "in": { "D": 0, "__clk": "rise" }, "out": { "Q": 0 } }
]
```

The level runner interprets `__clk` as: `"rise"` = low tick then high tick then read; `"stay"` = two ticks at current clk, no change. Adjust `Levelrun.run` to support labeled inputs (object form) in addition to array form.
